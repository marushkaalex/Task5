CREATE TABLE user(
id INT AUTO_INCREMENT,
username VARCHAR(255),
email VARCHAR(255),
password VARCHAR(255),
role INT,
dob DATE,
PRIMARY KEY (id),
FOREIGN KEY (role) REFERENCES role(id));
